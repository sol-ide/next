This todo section shows some of the features of Next.Event that should be implemented.

Get an handler when registering listeners 

next::dispatcher d{ next::thread_pool_size_t{ 4 } };

next::event_handler h{ d };

next::listener_handle l =
  h.listen< mine::is_less_than_ten >(
  []( int x ) -> bool
  {
    return x < 10;
  }
);

h.remove_listener( l );

Rewrite sending event with std::bind and std::function

Add the possibility to give a source to event 

next::dispatcher d{ next::thread_pool_size_t{ 4 } };

next::event_handler h_send{ d };

next::event_handler h_receive{ d };

h_receive.listen< mine::is_less_than_ten >(
  []( int x ) -> bool
  {
    return x < 10;
  }
);

d.send_event< mine::is_less_than_ten >( 5 ).from( h_send ).to( h_receive );
